\medskip
\textbf{1. Description of $L(A^{(1)})$ and $L(A^{(2)})$.}

\medskip
\underline{$A^{(1)}$:}

\begin{center}
\begin{tikzpicture}[shorten >=1pt,node distance=2.5cm,on grid,auto]
\node[state,initial] (1) {$1$};
\node[state,accepting] (2) [right=of 1] {$2$};
\node[state] (3) [right=of 2] {$3$};
\node[state,accepting] (4) [below=of 2] {$4$};

\path[->]
(1) edge node {$a$} (2)
    edge node[left] {$b$} (4)
(2) edge node {$a$} (3)
    edge node[right] {$b$} (4)
(3) edge[loop above] node {$a,b$} ()
(4) edge[bend left=15] node {$a$} (2)
    edge node[below left] {$b$} (3);
\end{tikzpicture}
\end{center}

The accepting states are $\{2,4\}$ and state $3$ is a non-accepting sink (it loops on both $a$ and $b$).
From state $2$, reading $a$ goes to the sink; from state $4$, reading $b$ goes to the sink.
Thus, once the first symbol is read, the letters must alternate in order to avoid the sink.

Therefore,
\[
L(A^{(1)})=\{\,w\in\{a,b\}^* : |w|\ge 1 \text{ and $w$ contains no substring $aa$ or $bb$}\,\}.
\]

Equivalently,
\[
L(A^{(1)}) = a(ba)^*(\varepsilon \mid b)\ \cup\ b(ab)^*(\varepsilon \mid a).
\]

\medskip
\underline{$A^{(2)}$:}

\begin{center}
\begin{tikzpicture}[shorten >=1pt,node distance=2.5cm,on grid,auto]
\node[state,initial] (1) {$1$};
\node[state,accepting] (2) [right=of 1] {$2$};
\node[state] (3) [below=of 1] {$3$};

\path[->]
(1) edge node {$a$} (2)
    edge node[left] {$b$} (3)
(2) edge[loop above] node {$a,b$} ()
(3) edge[loop below] node {$a,b$} ();
\end{tikzpicture}
\end{center}

The only accepting state is $2$. From state $1$, reading $b$ leads to a sink.
Reading $a$ moves $1 \to 2$, and from $2$ any symbol returns to $1$.

Thus a word must:
\begin{itemize}
\item start with $a$,
\item have odd length,
\item have $a$ in every odd position.
\end{itemize}

Hence,
\[
L(A^{(2)}) = a\big((a\mid b)a\big)^*.
\]

\medskip
\textbf{2. Intersection automaton $A = A^{(1)} \times A^{(2)}$.}

\begin{itemize}
\item States: $Q = Q_1 \times Q_2$
\item Start state: $(1,1)$
\item Accepting states:
\[
F = F_1 \times F_2 = \{2,4\}\times\{2\}
\]
\item Transition function:
\[
\delta((p,q),x) = (\delta_1(p,x), \delta_2(q,x))
\]
\end{itemize}

The reachable accepting state is $(2,2)$.

\medskip
\textbf{3. Proof that $L(A)=L(A^{(1)})\cap L(A^{(2)})$.}

For any word $w$, by induction on prefixes, after reading $w$ the product automaton is in state
\[
(\delta_1(1,w),\delta_2(1,w)).
\]
Thus $A$ accepts $w$ iff both component automata accept $w$.
Therefore,
\[
L(A)=L(A^{(1)})\cap L(A^{(2)}).
\]

\medskip
\textbf{4. Construction of union automaton $A'$.}

Keep the same product states and transitions, but define the accepting set as
\[
F' = (F_1\times Q_2)\ \cup\ (Q_1\times F_2).
\]
Then $A'$ accepts whenever at least one component accepts, so
\[
L(A') = L(A^{(1)}) \cup L(A^{(2)}).
\]

\medskip
\textbf{3. Proof that $L(B)=L(B^{(1)})\cap L(B^{(2)})$.}

After reading any word $w$, the product automaton is in state
\[
(\delta_1(p_0,w),\delta_2(q_0,w)).
\]
Thus $B$ accepts $w$ iff both component automata accept $w$.
Therefore,
\[
L(B)=L(B^{(1)})\cap L(B^{(2)}).
\]

\medskip
\textbf{4. Construction of union automaton $B'$.}

Using De Morgan's law:
\[
L(B^{(1)})\cup L(B^{(2)})
=
\overline{\ \overline{L(B^{(1)})}\ \cap\ \overline{L(B^{(2)})}\ }.
\]

\begin{enumerate}
\item Complement $B^{(1)}$ and $B^{(2)}$ by swapping accepting and non-accepting states.
\item Construct their product automaton for intersection.
\item Complement the resulting automaton.
\end{enumerate}

This yields $B'$ such that
\[
L(B') = L(B^{(1)}) \cup L(B^{(2)}).
\]


\paragraph{Exercise 2.2.7}
Let $A$ be a DFA and let $q$ be a state such that $\delta(q,a)=q$ for all input symbols $a\in\Sigma$.
We prove by induction on $|w|$ that for all strings $w\in\Sigma^*$, $\hat{\delta}(q,w)=q$.

\begin{proof}
\textbf{Base case:} $|w|=0$, so $w=\varepsilon$.
By definition of $\hat{\delta}$, $\hat{\delta}(q,\varepsilon)=q$.

\textbf{Inductive step:} Assume that for some $n\ge 0$, for every string $x\in\Sigma^*$ with $|x|=n$ we have $\hat{\delta}(q,x)=q$.
Let $w\in\Sigma^*$ be any string with $|w|=n+1$.
Then $w$ can be written as $w=xa$ where $|x|=n$ and $a\in\Sigma$ is the last symbol.
Using the recursive definition of $\hat{\delta}$,
\[
\hat{\delta}(q,xa)=\delta(\hat{\delta}(q,x),a).
\]
By the inductive hypothesis, $\hat{\delta}(q,x)=q$, so
\[
\hat{\delta}(q,xa)=\delta(q,a)=q,
\]
where the last equality uses the assumption that $\delta(q,a)=q$ for all $a\in\Sigma$.
Thus $\hat{\delta}(q,w)=q$ for all $w$ of length $n+1$.

By induction, $\hat{\delta}(q,w)=q$ for all $w\in\Sigma^*$.
\end{proof}

\paragraph{Exercise 1 -- Kleene's Algorithm}

Consider the DFA with states $p$ and $q$, where $p$ is the start and accepting state, and the transitions are:
\[
p \xrightarrow{a} p,\quad p \xrightarrow{b} q,\quad q \xrightarrow{a} p,\quad q \xrightarrow{b} q.
\]
Let $1=p$ and $2=q$.

For Kleene's algorithm, the base regular expressions are:
\[
R_{11}^{(0)}=\varepsilon + a,\qquad
R_{12}^{(0)}=b,\qquad
R_{21}^{(0)}=a,\qquad
R_{22}^{(0)}=\varepsilon + b.
\]

Using the recurrence
\[
R_{ij}^{(k)} = R_{ij}^{(k-1)} + R_{ik}^{(k-1)}\left(R_{kk}^{(k-1)}\right)^*R_{kj}^{(k-1)},
\]
we first compute the $k=1$ stage (eliminating state 1):
\begin{align*}
R_{11}^{(1)} &= (\varepsilon + a) + (\varepsilon + a)(\varepsilon + a)^*(\varepsilon + a) = a^* \\
R_{12}^{(1)} &= b + (\varepsilon + a)(\varepsilon + a)^* b = a^*b \\
R_{21}^{(1)} &= a + (\varepsilon + b)(\varepsilon + a)^* a = aa^* = a^+ \\
R_{22}^{(1)} &= (\varepsilon + b) + (\varepsilon + b)(\varepsilon + a)^* b = \varepsilon + a^*b
\end{align*}

Then the $k=2$ stage (eliminating state 2) gives the language expression $R_{11}^{(2)}$, since state 1 ($=p$) is the only accepting state:
\begin{align*}
R_{11}^{(2)} &= R_{11}^{(1)} + R_{12}^{(1)}\left(R_{22}^{(1)}\right)^* R_{21}^{(1)} \\
             &= a^* + a^*b\,(\varepsilon + a^*b)^*\,a^+ \\
             &= a^* + a^*b(a^*b)^*a^+
\end{align*}

This simplifies to the regular expression:
\[
R = \varepsilon + (a+b)^*a.
\]

Thus, the language accepted by the DFA is:
\[
L(A)=L(R)=\{\varepsilon\}\cup\{w\in\{a,b\}^* : w \text{ ends in } a\}.
\]

\subsection{Week 5}

\subsubsection*{Pumping Lemma Exercises}

We use the pumping lemma to show that each of the following languages is not regular. Recall that the pumping lemma states that if a language $L$ is regular, then there exists a pumping length $p$ such that any string $s \in L$ with $|s| \ge p$ can be written as $s = xyz$ where:

\begin{itemize}
\item $|xy| \le p$
\item $|y| > 0$
\item $xy^iz \in L$ for all $i \ge 0$
\end{itemize}

We will assume each language is regular and derive a contradiction.

\paragraph{$L_1 = \{a^n b^m \mid n,m \in \mathbb{N}, n > m \ge 0\}$}

Assume $L_1$ is regular and let $p$ be the pumping length.  
Choose the string:

\[
s = a^{p+1}b^p
\]

Clearly $s \in L_1$ because $n=p+1$ and $m=p$ so $n>m$.

By the pumping lemma, $s = xyz$ where $|xy| \le p$ and $|y|>0$. Since the first $p$ characters are all $a$'s, the substring $y$ must consist only of $a$'s.

Let $y = a^k$ where $k>0$.

Now pump down with $i=0$:

\[
xz = a^{p+1-k}b^p
\]

Now the number of $a$'s is $p+1-k$ which is $\le p$. This means the number of $a$'s is no longer greater than the number of $b$'s. Therefore $xz \notin L_1$.

This contradicts the pumping lemma, so $L_1$ is not regular.

\paragraph{$L_2 = \{a^{n^2} \mid n \ge 1\}$}

Assume $L_2$ is regular and let $p$ be the pumping length.

Choose the string:

\[
s = a^{p^2}
\]

Since $p^2$ is a perfect square, $s \in L_2$.

Because $|xy| \le p$, the substring $y$ consists only of $a$'s. Let $y = a^k$ with $k>0$.

Pump the string with $i=2$:

\[
xy^2z = a^{p^2 + k}
\]

However, $p^2 + k$ is not guaranteed to be a perfect square since $1 \le k \le p$. Therefore the pumped string is not of the form $a^{n^2}$.

Thus $xy^2z \notin L_2$, contradicting the pumping lemma. Therefore $L_2$ is not regular.

\paragraph{$L_3 = \{a^p \mid p \text{ is prime}\}$}

Assume $L_3$ is regular and let $p$ be the pumping length.

Choose a prime number $q > p$ and let:

\[
s = a^q
\]

Clearly $s \in L_3$.

Again $|xy| \le p$ so $y = a^k$ for some $k>0$.

Pump the string with $i=q+1$:

\[
xy^{q+1}z = a^{q + qk}
\]

The exponent becomes:

\[
q + qk = q(1+k)
\]

This number is composite since it has factors $q$ and $(1+k)$ with $k \ge 1$. Therefore the pumped string is not prime length.

Thus $xy^{q+1}z \notin L_3$, contradicting the pumping lemma. Therefore $L_3$ is not regular.

\paragraph{$L_4 = \{a^k b^m a^{k+m} \mid k,m \ge 0\}$}

Assume $L_4$ is regular and let $p$ be the pumping length.

Choose:

\[
s = a^p b^p a^{2p}
\]

This string belongs to $L_4$ since $k=p$ and $m=p$, so the final block has length $k+m = 2p$.

Because $|xy| \le p$, the substring $y$ lies entirely within the first block of $a$'s.

Let $y = a^j$ where $j>0$.

Pump down with $i=0$:

\[
xz = a^{p-j} b^p a^{2p}
\]

Now the first block has $p-j$ $a$'s while the final block still has $2p$ $a$'s. 
However the language requires the final block to have exactly $k+m$ $a$'s, 
which should now be:

\[
(p-j) + p = 2p - j
\]

But the final block still has $2p$ $a$'s, so the string no longer satisfies the language definition.

Thus $xz \notin L_4$, contradicting the pumping lemma. Therefore $L_4$ is not regular.

\begin{enumerate}
    \item \textbf{True.} Given deciders $M_1$ for $L_1$ and $M_2$ for $L_2$, construct a TM $M$ that on input $w$: runs $M_1$ on $w$; if $M_1$ accepts, accept; otherwise run $M_2$ on $w$ and return its answer. Both $M_1$ and $M_2$ always halt, so $M$ always halts. $M$ accepts $w$ iff $w\in L_1$ or $w\in L_2$, i.e.\ iff $w\in L_1\cup L_2$.

    \item \textbf{True.} Given a decider $M$ for $L$, construct $\overline{M}$ that runs $M$ and swaps accept/reject. Since $M$ always halts, so does $\overline{M}$, and it accepts exactly the strings not in $L$, i.e.\ $\overline{L}$.

    \item \textbf{True.} $\varepsilon\in L^*$ always, so accept immediately on empty input. For a non-empty input $w$, try all $2^{|w|-1}$ ways to partition $w = w_1 w_2 \cdots w_k$ and for each partition run the decider for $L$ on every piece. Accept if some partition has every piece in $L$; reject otherwise. There are finitely many partitions and the decider for $L$ always halts, so this procedure always terminates.

    \item \textbf{True.} Given recognizers $M_1$ for $L_1$ and $M_2$ for $L_2$, construct a TM $M$ that on input $w$ \emph{dovetails} (interleaves) the simulations: run $M_1$ for one step, then $M_2$ for one step, then $M_1$ for two steps, and so on. If either machine ever accepts, $M$ accepts. Since any $w\in L_1\cup L_2$ is accepted by at least one of $M_1, M_2$ in finitely many steps, $M$ will eventually accept it. Hence $L_1\cup L_2$ is r.e.

    \item \textbf{False.} The halting language $\mathit{HALT} = \{(M,w)\mid M \text{ halts on } w\}$ is r.e.\ (simulate $M$ on $w$; accept if it halts) but its complement $\overline{\mathit{HALT}}$ is not r.e.\ This is a standard result: if both $\mathit{HALT}$ and $\overline{\mathit{HALT}}$ were r.e., then $\mathit{HALT}$ would be decidable, contradicting the undecidability of the halting problem.

    \item \textbf{True.} Given a recognizer $M$ for $L$, we build a recognizer for $L^*$. On input $w$, enumerate all partitions of $w$ into non-empty pieces and dovetail the simulation of $M$ on each piece across all partitions simultaneously. Accept as soon as all pieces in some partition are accepted by $M$. Since any $w\in L^*$ has a valid decomposition into words of $L$, each accepted in finite time, the machine will eventually find and confirm that partition.
\end{enumerate}


\paragraph{Exercise 2 -- Classification}

\begin{enumerate}
    \item $L_1 = \{M \mid M \text{ halts on itself}\}$

    \begin{itemize}
        \item r.e.: yes
        \item decidable: no
        \item co-r.e.: no
    \end{itemize}

    \item $L_2 = \{(M,w) \mid M \text{ halts on } w\}$

    \begin{itemize}
        \item r.e.: yes
        \item decidable: no
        \item co-r.e.: no
    \end{itemize}

    \item $L_3 = \{(M,w,k) \mid M \text{ halts in } \le k \text{ steps}\}$

    \begin{itemize}
        \item decidable: yes
        \item r.e.: yes
        \item co-r.e.: yes
    \end{itemize}
\end{enumerate}
